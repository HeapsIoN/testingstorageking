//%attributes = {"lang":"en"} comment added and reserved by 4D.
$Find:=<>sFacility+[UnitType]UnitTypeCode
If (Find in field([UnitType]RUnitType;$Find)>0)
AlertMessage (Get localized string("UnitType")+" "+[UnitType]UnitTypeCode+" "+Get localized string("already exists. Unable to change!"))
[UnitType]UnitTypeCode:=Old([UnitType]UnitTypeCode)
Else 
QUERY([Unit];[Unit]UnitTypeCode=Old([UnitType]RUnitType))
[UnitType]RUnitType:=Substring([UnitType]RUnitType;1;5)+[UnitType]UnitTypeCode
If (Records in selection([Unit])>0)
ConfirmTwo (Get localized string("Change")+" "+String(Records in selection([Unit]))+" "+Get localized string("Units with UnitType")+" "+Old([UnitType]RUnitType)+" "+"to1"+" "+[UnitType]RUnitType+"?")
If (OK=1)
READ WRITE([Unit])
FIRST RECORD([Unit])
APPLY TO SELECTION([Unit];[Unit]UnitTypeCode:=[UnitType]RUnitType)
READ ONLY([Unit])
UNLOAD RECORD([Unit])

//23/12/16 DM Fixed Rent Increase history not updateing in a case UnitType is changed
QUERY([RCHistory];[RCHistory]UnitType=Old([UnitType]RUnitType))
If (Records in selection([RCHistory])>0)
READ WRITE([RCHistory])
APPLY TO SELECTION([RCHistory];[RCHistory]UnitType:=[UnitType]RUnitType)
READ ONLY([RCHistory])
UNLOAD RECORD([RCHistory])
End if 

ACCEPT
VALIDATE TRANSACTION
bCancel:=1
Else 
[UnitType]UnitTypeCode:=Old([UnitType]UnitTypeCode)
[UnitType]RUnitType:=Old([UnitType]RUnitType)
End if 
End if 
End if
