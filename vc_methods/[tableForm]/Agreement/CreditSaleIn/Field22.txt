Else

Else

AlertMessage (Get localized string("WARNING - The Next Bill Date should not be earlier than today."))
End if

AlertMessage (Get localized string("WARNING - You have changed the Next Bill Date by more than one month. Please check this carefully..."))
End if 
End if
